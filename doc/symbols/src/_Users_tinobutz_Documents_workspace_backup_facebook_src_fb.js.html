185     /**
186      * The tree class represents an interval tree based on a red / black tree.
187      * An interval tree is the optimal data structore for events. Insert
188      * is O(lg n) and (as the tree is balanced) search for a node can be achieved
189      * in O(lg n) time as well. See Cormen, introduction to Algorithms, Interval Trees
190      * for more information.
191      *
192      * @constructor
193      * @param {Object} comparator The comperator delegate. The delgate gets called whenever a node is added and two nodes have to be compared.
194      *                            Returns 0 when the start time of the first event is less than the start time of the second event. 1 for the opposite case.
195      */
196     FB.data.Tree = function(comparator) {
197     this._root = null;
198     this._comparator = comparator;
199     this._size = 0;
200   };

213   /**
214    * Inserts a node into the tree.
215    * 
216    * @param {Object} data The data assoicated with the new node.  All methods of the interval interface have to be implemented (id / start / end).
217    */  
218   Tree.prototype.insert = function(data) {
219     
220     var node = this._insertNode(data);
221   
222     // Set the maximum for the interval tree
223     node.setMax(this._getMax(node));
224     
225     while (node != this._root && node.getParent().isRed()) {
226       var direction = Node.LEFT;
227       if (node.getParent() == node.getGrandfather().getChild(Node.LEFT)) {
228         direction = Node.RIGHT;
229       }
230       var node = this._checkViolation(node, direction);
231     }
232   
233     // Root is always black
234     this._root.setColor(Node.BLACK);	
235   };

238   /**
239    * Checks for violations of the constraints of a red / black tree.
240    * Left / Right rotates a node, when a constraint is violated.
241    * 
242    * @param {FB.data.Node} node The node to check the violation for.
243    * @param {Integer} direction Node.LEFT or Node.RIGHT.
244    * @returns {FB.data.Node} The right node after the check.
245    */
246   Tree.prototype._checkViolation = function(node, direction) {
247       // Check violations
248       var uncle = node.getGrandfather().getChild(direction);
249       if (uncle && uncle.isRed()) {
250         node.getParent().setColor(Node.BLACK);
251         uncle.setColor(Node.BLACK);
252         node.getGrandfather().setColor(Node.RED);
253         node = node.getGrandfather();
254       } else {
255         if (node == node.getParent().getChild(direction)) {
256           node = node.getParent();
257           if (direction === Node.RIGHT) {
258             this._leftRotate(node);
259           } else {
260             this._rightRotate(node);        
261           }
262         }
263         node.getParent().setColor(Node.BLACK);
264         node.getGrandfather().setColor(Node.RED);
265         if (direction === Node.RIGHT) {
266           this._rightRotate(node.getGrandfather());
267         } else {
268           this._leftRotate(node.getGrandfather());        
269         }
270       }
271       return node;
272   };

275   /**
276    * Returns the maximum of the interval of the node or its children. Used for optimization
277    * for the interval search. Sub trees that does not match the maximum and the
278    * searched interval can be skiped.
279    *
280    * @param {FB.data.Node} node The node to get the max for.
281    * @returns {Integer} The maximum of the interval or its children.
282    */
283   Tree.prototype._getMax = function(node) {
284     var left = node.getChild(Node.LEFT);
285     var right = node.getChild(Node.RIGHT);
286     return Math.max(node.data.end, Math.max(left != null ? left.getMax() : 0, right != null ? right.getMax() : 0));
287   };

310   /**
311    * Rotates a node to a certain direction. Used to keep the balance of the red / black tree.
312    *
313    * @param {FB.data.Node} node The node to rotate.
314    * @param {Integer} direction Node.LEFT or Node.RIGHT.
315    */ 
316   Tree.prototype._rotate = function(node, direction) {
317   
318     var otherDirection = (direction === Node.RIGHT ? Node.LEFT : Node.RIGHT);
319   
320     var child = node.getChild(otherDirection);
321     if (child === null) throw new Error("Child is null");
322     
323     node.setChild(child.getChild(direction), otherDirection);
324     if (child.getChild(direction) != null) { 
325       child.getChild(direction).setParent(node);
326     }
327     child.setParent(node.getParent());
328     if (node.getParent() === null) {
329       this._root = child;	
330     } else {
331       if (node == node.getParent().getChild(direction)) {
332         node.getParent().setChild(child, direction);
333       } else {
334         node.getParent().setChild(child, otherDirection);
335       }
336     }
337     child.setChild(node, direction);
338     node.setParent(child);
339     
340     // Update the max
341     child.setMax(this._getMax(child));
342     node.setMax(this._getMax(node));	
343   };

346   /**
347    * Inserts a node to the tree and stores the given data.
348    *
349    * @param {Object} data The data assoicated with the new node.
350    * @returns {FB.data.Node} The added node.
351    */
352   Tree.prototype._insertNode = function(data) {
353     
354     var newNode = new Node(data); // z
355   
356     var parent = null; // y
357     var node = this._root; // x
358     while (node != null) {
359       parent = node;
360       node = node.getChild(this._comparator(newNode.data, node.data));
361     }
362     
363     newNode.setParent(parent);
364   
365     if (parent === null) {
366       this._root = newNode;
367     } else {
368         parent.setChild(newNode, this._comparator(newNode.data, parent.data));			
369     }
370     
371     this._size++;
372     
373     return newNode;
374   };

411   /**
412    * Returns the successor node. When no node is passed as an argument,
413    * search begins from root.
414    *
415    * @param {FB.data.Node} [node] The node to start the search from.
416    * @returns {FB.data.Node} The successor node.
417    */
418   Tree.prototype.successor = function(node) {
419     if (node.getChild(Node.RIGHT) != null) {
420       return this.minimum(node.getChild(Node.RIGHT));	
421     }
422     var parent = node.getParent();
423     while (parent && node == parent.getChild(Node.RIGHT)) {
424       var node = parent;
425       parent = parent.getParent();	
426     }
427     return parent;
428   };

508   /**
509    * Returns all intervals which collide with the given interval.
510    * 
511    * @param {Object} interval The interval to search for in the tree.  All methods of the interval interface have to be implemented (id / start / end).
512    * @param {FB.data.Node} [node] The node to start the search from.
513    * @param {Object[]} The colliding intervals.
514    */
515   Tree.prototype.searchInterval = function(interval, node) {
516        node = node || this._root;
517        result = [];
518   
519        var left = node.getChild(Node.LEFT);
520        if (left != null && left.getMax() >= interval.start) {
521          result = result.concat(this.searchInterval(interval, left));
522        }
523        
524        if (Event.isColliding(node.data, interval)) {
525          result.push(node.data);
526        }
527        
528        var right = node.getChild(Node.RIGHT);
529        if (right != null && node.data.start <= interval.end) {
530          result = result.concat(this.searchInterval(interval, right));
531        }
532        
533        return result;
534   };

539    * When no node instance is passed as an argument, the root of the tree
540    * is used to calculate the black height.
541    *
542    * @param {FB.data.Node} [node] Optional node to start from.

546     node = node || this._root;
547     var height = 0;
548     while (node != null) {
549       node = node.getChild(0);
550       if (node === null || node.isBlack()) { // nil is always black

552       }
553     }

573   /**
574    * Converts the tree to a valid Dot string. This string can be used to 
575    * display the tree in a Dot viewer. Usefull for debugging.
576    *
577    * @returns {String} The Dot string.
578    */
579   Tree.prototype.toDot = function() {
580     var nodes = this.getOrdered();
581     var dot = [];
582     dot.push("digraph G {");
583   
584     for(var i = 0, j = nodes.length; i < j; i++) {
585       var node = nodes[i];
586       var color = node.isRed() ? "red" : "black";
587       var font =  "white";
588       dot.push(node.data.id + ' [label="' + node.data.id + ': [' + node.data.start + ',' +  node.data.end + '] (Max: ' + node.getMax() + ')",fontcolor=' + font + ',fillcolor=' + color + ',style=filled];');
589       if (node.getParent()) {
590         dot.push(node.getParent().data.id + " -> " + node.data.id+ ";");
591       }
592     }
593   
594     dot.push("}");
595     return dot.join("\n");	
596   };

1030   /**
1031    * Protected render method. Used internally to render all events in a recursion.
1032    * 
1033    * @param {FB.ui.calendar.Event[]} events The events to render in this roundtrip.
1034    * @param {Object[]} binsByRef The bins containing all collding events.
1035    */
1036   Calendar.prototype._render = function(events, binsByRef) {
1037     var html = [];
1038   
1039     var length = events.length;
1040     for (var i=0; i < length ; i++)
1041     {
1042       var event = events[i];
1043       var bins = binsByRef || [];
1044       if (!this.isRendered(event))
1045       {
1046         this._setIsRendered(event);
1047         this._addToBin(event, bins);
1048   
1049         var collisions = this.getCollisions(event);
1050         if (collisions) {
1051           html = html.concat(this._render(collisions, bins));    
1052         }
1053   
1054         html = html.concat(this._renderEvents(bins));
1055         }
1056     }
1057     return html;
1058   };

1061   /**
1062    * Renders the events.
1063    *
1064    * @param {Object[]} bins The bins containing all collding events.
1065    */
1066   Calendar.prototype._renderEvents = function(bins) {
1067     var html = [];
1068     for (var level = 0; level < bins.length; level++) {
1069       var bin = bins[level];
1070       var events = bin.events;
1071       for (var i = 0; i < events.length; i++) {
1072         var width = this._options.width / (bins.length);
1073         html.push(events[i].render(width, level));
1074       }
1075     }
1076     return html;	
1077   };

1080   /**
1081    * Adds an event to the first free bin. When the start time is greater or 
1082    * equal the maximum of the bin, the event is added. A bin is representing
1083    * a time slot and has a maximum, which equals the end time of the last
1084    * added event.
1085    *
1086    * @param {FB.ui.calendar.Event[]} event The event to add to a bin.
1087    * @param {Object[]} bins All existing bins.
1088    */
1089   Calendar.prototype._addToBin = function(event, bins) {
1090     var length = bins.length;
1091     
1092     var added = false;
1093     for (var i=0; i < length; i++)
1094     {
1095       var bin = bins[i];
1096       if (event.start >= bin.max) {
1097         bin.max = event.end;
1098         bin.events.push(event);
1099         added = true;
1100         break;
1101       }
1102     }
1103     if (!added) {
1104       bins.push({
1105         max : event.end,
1106         events : [event]
1107       });
1108     }
1109   };
